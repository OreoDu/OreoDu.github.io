1.Two Sum

Keyword ：Search, Hash, Map

def twoSum(nums, target):
    h = {}
    for i,num in enumerate(nums):
        if target - num in h:
            return [h[target - num], i]
        h[num] = i

#T(n) = O(n)

167.Two Sum Ⅱ - Input array is sorted

Keyword: Double pointer

def twoSum(numbers, target):
    lens = len(numbers)
    i = 0
    j = lens-1
    while i<j:
        if numbers[i]+numbers[j] > target:
            j-=1
        elif numbers[i]+numbers[j] < target:
            i+=1
        else:
            return [i+1,j+1]
